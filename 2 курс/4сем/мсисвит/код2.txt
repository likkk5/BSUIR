// Example of a while loop
def outerCount = 0
while (outerCount < 3) {
  println("Outer iteration $outerCount")
  def innerCount = 0
  while (innerCount < 2) {
    println("  Inner iteration $innerCount")
    innerCount++
  }
  outerCount++
}

// Example of nested loops and conditional statements
for (i in 1..3) {
  println("Outer loop, iteration $i")
  for (j in 1..2) {
    println("Inner loop, iteration $j")
    if (j == 2) {
      println("Condition inside inner loop")
    }
  }
}
// Creating a two-dimensional matrix
def matrixA = [
        [0, 0, 0],
        [0, 0, 0],
        [0, 0, 0]
]

// Generating random numbers and filling the matrix with them
def rowIndex = 0
do {
  def colIndex = 0
  do {
    matrixA[rowIndex][colIndex] = new Random().nextInt(10) // Generating a random number from 0 to 9
    colIndex++
  } while (colIndex < matrixA[rowIndex].size())
  rowIndex++
} while (rowIndex < matrixA.size())

// Displaying the matrix
println "Generated matrix:"
for (row in matrixA) {
  println row
}
