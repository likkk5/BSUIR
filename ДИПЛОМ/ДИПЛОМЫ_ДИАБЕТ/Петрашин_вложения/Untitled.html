def p(t, d):
    if (t >= 0 and t <= d / 2):
        return 4 * t / d ** 2
    if ((t >= d / 2) and (t <= d)):
        return 4 / d - 4 * t / d ** 2
    return 0

def v1(t):
    part_1 = patient_data['m1'] * p(t - patient_data['t1'], patient_data['d1'])
    part_2 = patient_data['m2'] * p(t - patient_data['t2'], patient_data['d2'])
    part_3 = patient_data['m3'] * p(t - patient_data['t3'], patient_data['d3'])
    return part_1 + part_2 + part_3

def v2(t):
    part_1 = patient_data['m4'] * p(t - patient_data['t4'], patient_data['d4'])
    part_2 = patient_data['m5'] * p(t - patient_data['t5'], patient_data['d5'])
    return part_1 + part_2

def v(t):
    return v1(t) + v2(t)

def D(t):
    return 1000 * v(t) / 180

def FR(t, z):
    if (z[0] / (patient_data['BW'] * patient_data['VG']) >= 9):
        return 0.003 * (z[0] / (patient_data['BW'] * patient_data['VG']) - 9) * patient_data['VG'] * patient_data['BW']
    return 0

def F01C(t, z):
    if (z[0] / (patient_data['BW'] * patient_data['VG']) >= 4.5):
        return patient_data['F01'] * patient_data['BW']
    return patient_data['F01'] * z[0] / (4.5 * patient_data['VG'])

def FQ1(t, z):
    part_1 = UG(t) - F01C(t, z) - FR(t, z) - z[2] * z[0]
    return part_1 + patient_data['BW'] * patient_data['EGPO'] * (1 - z[4]) + patient_data['K12'] * z[1]
